# mysql 查询带有逗号分隔的字段 find_in_set

 


mysql 自带方法 find_in_set('条件', 字段);

表数据

```
+----+-------+------------+

| id | name  | pid        |

+----+-------+------------+

|  1 | name  | 1,2,3,4,5  |

|  2 | pname | 1,2,3,44,5 |

+----+-------+------------+
```

```
select * from test where find_in_set('3', pid);
```

```
+----+------+-----------+

| id | name | pid       |

+----+------+-----------+

|  1 | name | 1,2,3,4,5 |

+----+------+-----------+
```

```
select * from test where find_in_set('44', pid) or find_in_set('5', pid);
```

```
+----+-------+------------+

| id | name  | pid        |

+----+-------+------------+

|  1 | name  | 1,2,3,4,5  |

|  2 | pname | 1,2,3,44,5 |

+----+-------+------------+
```
